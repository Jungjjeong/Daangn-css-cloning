│   └── style
│        │── _sass
│        │     │── _header.scss 
│        │     │── _home-main.scss

│        │     │── _footer.scss 
│        │     │── _variables.scss - 변수
│        │     │── _mixins.scss - mixin 규칙
│        │     │── _print.scss - 프린트되는 글자 태그들의 옵션
│        │     │── _common.scss - 맨 처음의 *, body 등 태그 초기화

│        └── css
